**Goal:** Add GM/T 0003.3 SM2 key agreement (with key confirmation) to `gmcrypto-core` behind an opt-in `sm2-key-exchange` feature, byte-identical to the GB/T 32918.3 Annex worked example.

**Architecture:** A new `sm2/key_exchange.rs` module exposing two role state-machines (`Sm2KxInitiator` / `Sm2KxResponder`) that reuse the existing SM2 curve arithmetic, masked nonce sampler, SM3 KDF, `compute_z`, and point-validation helpers. **Tech Stack:** Rust 2024 / MSRV 1.85, `no_std` + `alloc`, `crypto-bigint 0.7` (`Fn`/`Fp` Montgomery forms), `subtle` (constant-time), `zeroize`, `rand_core::TryCryptoRng`.

use crate::sm2::curve::Fn;
use crypto_bigint::U256;

/// avf(x) = 2^127 + (x mod 2^127), per GB/T 32918.3. `x_be` is the affine
/// x-coordinate of R as a 32-byte big-endian integer. Constant-time: pure
/// bit masking, no branch on `x`. Result < 2^128 < n, so `Fn::new` is an
/// identity reduction.
fn avf(x_be: &[u8; 32]) -> Fn {
    let mut buf = [0u8; 32];
    // Keep the low 127 bits: bytes 17..32 in full (120 bits) + low 7 bits of byte 16.
    buf[17..32].copy_from_slice(&x_be[17..32]);
    buf[16] = (x_be[16] & 0x7F) | 0x80; // low 7 bits + set bit 127
    Fn::new(&U256::from_be_slice(&buf))
}
```

/// Fixed-budget masked ephemeral: sample r âˆˆ [1,nâˆ’1] via the existing
/// `sample_nonzero_scalar`, compute R = [r]G, return (r, SEC1 R). RNG failure â†’ Failed.
fn sample_ephemeral<R: TryCryptoRng>(rng: &mut R) -> Result<(crate::sm2::curve::Fn, [u8; 65])> {
    use crate::sm2::sign::{sample_nonzero_scalar, SIGN_RETRY_BUDGET};
    use subtle::ConditionallySelectable;
    let mut acc_r = crate::sm2::curve::Fn::new(&crypto_bigint::U256::ONE);
    let mut got = Choice::from(0);
    for _ in 0..SIGN_RETRY_BUDGET {
        let (cand, ok) = sample_nonzero_scalar(rng).ok_or(Error::Failed)?;
        let take = ok & !got;
        acc_r = crate::sm2::curve::Fn::conditional_select(&acc_r, &cand, take);
        got |= ok;
    }
    if !bool::from(got) { return Err(Error::Failed); }
    let r_point = mul_g(&acc_r);
    let (x, y) = r_point.to_affine().ok_or(Error::Failed)?;
    let mut sec1 = [0u8; 65];
    sec1[0] = 0x04;
    sec1[1..33].copy_from_slice(&u256_to_be32(&x.retrieve()));
    sec1[33..65].copy_from_slice(&u256_to_be32(&y.retrieve()));
    Ok((acc_r, sec1))
}
```

```rust
/// Compute the agreed key bytes + (x_U, y_U) for the S-tag hashes.
/// `d` = local static secret, `r_eph` = local ephemeral secret,
/// `r_local_x` = avf input from the LOCAL R, `peer_r` = peer's SEC1 R,
/// `p_peer` = peer static point. Returns Failed on off-curve R or identity U.
fn shared_secret(
    d: &crate::sm2::curve::Fn, r_eph: &crate::sm2::curve::Fn,
    r_local_x: &[u8; 32], peer_r: &[u8; 65], p_peer: &ProjectivePoint,
    z_a: &[u8; 32], z_b: &[u8; 32], klen: usize,
) -> Result<(Vec<u8>, [u8; 32], [u8; 32])> {
    // Validate + parse peer R (on-curve check is the invalid-curve defense).
    let peer_pub = Sm2PublicKey::from_sec1_bytes(peer_r).ok_or(Error::Failed)?;
    let peer_point = peer_pub.point();
    let peer_x = {
        let s = peer_pub.to_sec1_uncompressed();
        let mut x = [0u8; 32]; x.copy_from_slice(&s[1..33]); x
    };

    let x_bar_local = avf(r_local_x);          // x̀„ of LOCAL R
    let x_bar_peer  = avf(&peer_x);            // x̀„ of PEER R

    // t = (d + x̀„_local Â· r_eph) mod n     (h = 1 for SM2)
    let t = *d + x_bar_local * *r_eph;
    // U = [t]( P_peer + [x̀„_peer] R_peer )
    let sum = p_peer.add(&mul_var(&x_bar_peer, &peer_point));
    let u = mul_var(&t, &sum);
    let (xu, yu) = u.to_affine().ok_or(Error::Failed)?;   // None == U is identity â†’ Failed
    let xu_b = u256_to_be32(&xu.retrieve());
    let yu_b = u256_to_be32(&yu.retrieve());

    // K = KDF(x_U || y_U || Z_A || Z_B, klen); reject all-zero K.
    let mut kin = Vec::with_capacity(64 + 64);
    kin.extend_from_slice(&xu_b); kin.extend_from_slice(&yu_b);
    kin.extend_from_slice(z_a);   kin.extend_from_slice(z_b);
    let mut key = alloc::vec![0u8; klen];
    crate::sm2::encrypt::kdf(&kin, &mut key);
    kin.zeroize();
    let mut allzero = Choice::from(1);
    for b in &key { allzero &= b.ct_eq(&0u8); }
    if bool::from(allzero) { return Err(Error::Failed); }
    Ok((key, xu_b, yu_b))
}
```
